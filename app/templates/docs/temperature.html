1) Overview

This firmware reads temperature and relative humidity from a DHT11 sensor on an ESP32, then issues two back‑to‑back HTTP POST requests once per minute to a Flask endpoint on the local network. Each POST contains one measurement:

  • Temperature as either Celsius (dht11_temp_c) or Fahrenheit (dht11_temp_f) depending on USE_FAHRENHEIT.
  • Humidity percentage as dht11_humidity_pct.

The route persists records into a database table readings(sensor_type, plant_id, reading).

  • Avoid long, noisy cables; keep DATA run short to reduce timing errors.
  • USB 5 V → ESP32 onboard regulator → 3.3 V rail.

5) Payload Contract

The Flask route expects JSON keys shown below. plant_id is an integer in this build.

Temperature POST

{
  "sensor_type": "dht11_temp_f",   // or "dht11_temp_c" if USE_FAHRENHEIT=false
  "reading": 73.00,                // float
  "plant_id": 1                    // int
}

Humidity POST

{
  "sensor_type": "dht11_humidity_pct",
  "reading": 45.00,
  "plant_id": 1
}

Server response 200/201 indicates success; other codes are logged to Serial.

6) Build & Flash (Arduino IDE)

  1. Install ESP32 board support via Boards Manager (Espressif Systems).
  2. Libraries:
    • DHT sensor library by Adafruit.
    • (Optional) Adafruit Unified Sensor if prompted as a dependency.
  3. Board: ESP32 Dev Module (or your specific ESP32 variant).
  4. Set correct COM/tty port.
  5. Open the sketch and edit the User Config block:
    • WIFI_SSID, WIFI_PASS
    • SERVER_URL (defaults to your LAN host)
    • PLANT_ID (int)
    • DHT_PIN (GPIO for DATA)
    • USE_FAHRENHEIT (true/false)
  6. Compile and upload. Open Serial Monitor at 115200 baud.

7) Runtime Behavior

State Flow

  1. Boot: Start Serial → init DHT → connect Wi‑Fi.
  2. Cycle (once per minute):
    1. Read humidity, tempC, tempF.
    2. Validate (reject NaN, clamp humidity 0–100%).
    3. POST temperature JSON.
    4. POST humidity JSON.
    5. Hard delay 60 s.
  3. Connectivity: If Wi‑Fi drops, reconnect before next cycle.

Design Notes

  • DHT11 has ~1 Hz max sample rate. One read per minute is conservative and reliable.
  • Back‑to‑back posts ensure separate DB rows per measurement.
  • Client timeout 5 s avoids long stalls on server issues.

9) Full Sketch

/*
  ESP32 + DHT11 -> Flask JSON (back-to-back posts each minute)

  Route expects JSON:
    { "sensor_type": "<string>", "reading": <float>, "plant_id": <int> }

  Posts in this order each cycle:
    1) sensor_type = "dht11_temp_c" (or "_f" if USE_FAHRENHEIT = true)
    2) sensor_type = "dht11_humidity_pct"

  Library deps (Arduino Library Manager):
    - "DHT sensor library" by Adafruit
*/

#include <WiFi.h>
#include <HTTPClient.h>
#include <DHT.h>

// ========= USER CONFIG =========
const char* WIFI_SSID   = "4301b";   // e.g. "4301b"
const char* WIFI_PASS   = "scoutdog1";   // e.g. "scoutdog1"

// Flask route
const char* SERVER_URL  = "http://192.168.50.243:5001/add_data";

// Identify the plant in your DB
const int   PLANT_ID    = 1;

// DHT wiring and type
#define DHT_PIN   4        // <-- set to your GPIO
#define DHT_TYPE  DHT11    // DHT11 module

// Units
#define USE_FAHRENHEIT true   // true => post temp in °F with type "dht11_temp_f"

// Interval
constexpr unsigned long POST_INTERVAL_MS = 60UL * 1000UL;
// ===============================

// Sensor types for route
const char* SENSOR_TYPE_TEMP_C = "dht11_temp_c";
const char* SENSOR_TYPE_TEMP_F = "dht11_temp_f";
const char* SENSOR_TYPE_HUM    = "dht11_humidity_pct";

// Globals
DHT dht(DHT_PIN, DHT_TYPE);
unsigned long lastPostMs = 0;

// Forward decls
void connectWiFi();
void ensureWiFiConnected();
bool postReading(const char* sensorType, float value);
void doOneCycle();

void setup() {
  Serial.begin(115200);
  delay(150);
  Serial.println("\n[BOOT] ESP32 DHT11 -> Flask JSON");

  // Start sensors
  dht.begin();
  Serial.printf("[DHT] Init on pin %d, type DHT11\n", DHT_PIN);

  // Wi-Fi
  connectWiFi();

  // Trigger immediate first send
  lastPostMs = 0;
  Serial.println("[BOOT] Ready.");
}

void loop() {
  ensureWiFiConnected();

  unsigned long now = millis();
  if (lastPostMs == 0 || (now - lastPostMs) >= POST_INTERVAL_MS) {
    lastPostMs = now;
    doOneCycle();         // back-to-back posts
    // Explicit pause to match "back to back then pause 60s"
    // Note: we also guard via millis() above, but this enforces a hard wait.
    delay(POST_INTERVAL_MS);
  }
  delay(25);
}

void doOneCycle() {
  // Read once. Adafruit DHT does an internal sample and caches briefly.
  // DHT11 resolution is low; expect integers or 0.5 steps.
  float h = dht.readHumidity();                                  // %
  float tC = dht.readTemperature();                              // °C
  float tF = dht.readTemperature(true);                          // °F

  // Debug prints for raw reads
  Serial.printf("[READ] Raw -> humidity=%.1f%% tempC=%.1fC tempF=%.1fF\n", h, tC, tF);

  // Validate readings (NaN when read fails)
  if (isnan(h) || isnan(tC) || isnan(tF)) {
    Serial.println("[READ] DHT read failed (NaN). Skipping this cycle.");
    return;
  }

  // Clamp humidity to a sane range
  if (h < 0.0f) h = 0.0f;
  if (h > 100.0f) h = 100.0f;

  // Choose temp unit
  float tempValue = USE_FAHRENHEIT ? tF : tC;
  const char* tempType = USE_FAHRENHEIT ? SENSOR_TYPE_TEMP_F : SENSOR_TYPE_TEMP_C;

  // Back-to-back posts: temperature then humidity
  bool ok1 = postReading(tempType, tempValue);
  bool ok2 = postReading(SENSOR_TYPE_HUM, h);

  Serial.printf("[POST] temp=%s hum=%s\n", ok1 ? "OK" : "FAIL", ok2 ? "OK" : "FAIL");
}

void connectWiFi() {
  Serial.printf("[WIFI] Connecting to SSID: %s\n", WIFI_SSID);
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  uint32_t start = millis();
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    if (millis() - start > 20000) {
      Serial.println("\n[WIFI] Timeout. Re-attempting.");
      WiFi.disconnect(true);
      WiFi.begin(WIFI_SSID, WIFI_PASS);
      start = millis();
    }
  }
  Serial.printf("\n[WIFI] Connected. IP=%s\n", WiFi.localIP().toString().c_str());
}

void ensureWiFiConnected() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[WIFI] Disconnected. Reconnecting.");
    connectWiFi();
  }
}

bool postReading(const char* sensorType, float value) {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[HTTP] No Wi-Fi. Abort POST.");
    return false;
  }

  WiFiClient client;   // HTTP (no TLS)
  HTTPClient http;
  http.setTimeout(5000);

  if (!http.begin(client, SERVER_URL)) {
    Serial.println("[HTTP] http.begin() failed");
    return false;
  }

  http.addHeader("Content-Type", "application/json");

  // Build compact JSON manually
  String payload = String("{\"sensor_type\":\"") + sensorType +
                   "\",\"reading\":" + String(value, 2) +
                   ",\"plant_id\":" + String(PLANT_ID) + "}";

  Serial.print("[HTTP] POST payload: ");
  Serial.println(payload);

  int code = http.POST(payload);
  Serial.printf("[HTTP] Status: %d\n", code);
  String resp = http.getString();
  Serial.print("[HTTP] Response: ");
  Serial.println(resp);

  http.end();
  return (code == 201 || code == 200);
}

11) Troubleshooting

NaN Readings

  • Verify DATA pin and DHT_PIN match.
  • Ensure 10 kΩ pull‑up on DATA if using bare sensor.
  • Reduce cable length; keep under ~20 cm if possible.
  • Power the sensor from 3V3, not 5V, on ESP32.

Wi‑Fi Fails

  • Check SSID/password.
  • Confirm 2.4 GHz enabled on AP.
  • Observe IP printed after connect; ping from PC.

Server Unreachable

  • Open the Flask route URL from a browser on same LAN.
  • Check Flask logs for incoming requests or errors.
  • Test with curl -X POST -H "Content-Type: application/json" \ -d '{"sensor_type":"test","reading":1,"plant_id":1}' http://192.168.50.243:5001/add_data

Outliers

  • DHT11 accuracy: ±2 °C, ±5% RH typical; do not overfit.
  • Consider debouncing spikes server‑side if needed.

12) Validation Checklist

  • Serial shows successful Wi‑Fi connection.
  • Two HTTP 200/201 responses per minute.
  • Database rows appear with expected sensor_type values.
  • Humidity within 0–100%, temperature plausible for environment.

13) Security & Hardening

  • Plain HTTP on LAN. For wider networks, use TLS termination or a reverse proxy.
  • Add a shared token header (e.g., X-API-Key) and validate server‑side.
  • Rate‑limit on the Flask route if many devices send data.

14) Extensions & Variations

  • Switch to DHT22 for better accuracy and wider range.
  • Attach a status LED to show POST success/fail.
  • Buffer failed readings and retry with exponential backoff.
  • Include timestamps in payload and store server‑side.
  • Batch both readings in a single JSON if schema allows.
