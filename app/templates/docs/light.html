1) Overview

This firmware reads illuminance (lux) from a BH1750 ambient light sensor over I²C and posts one JSON record to a Flask route every 60 seconds. The sketch initializes I²C on SDA=21 and SCL=22, probes both standard BH1750 addresses (0x23 and 0x5C), performs a single high‑resolution measurement per cycle via readLightLevel(), and submits the lux value as a floating‑point number.

3) Wiring & Power

Connections

BH1750ESP32Notes
VCC3V3Use 3.3 V on ESP32. Many breakouts tolerate 5 V, but 3.3 V is safest.
GNDGNDCommon ground required.
SDAGPIO 21I²C data line.
SCLGPIO 22I²C clock line.
ADDRGND or 3V3GND → 0x23, 3V3 → 0x5C. Leave floating for 0x23 on many boards.

Most BH1750 modules include ~10 kΩ pull‑ups on SDA/SCL. If using a bare IC, add pull‑ups to 3V3.

Power Notes

  • Typical BH1750 current is ~0.12 mA active. Negligible versus ESP32.
  • Keep I²C runs short. Twisted SDA/GND and SCL/GND pairs help in noisy environments.
  • If multiple I²C devices share the bus, ensure a single effective pull‑up per line.

4) I²C Addressing & Modes

Addresses

  • 0x23: ADDR low/floating (default on most modules).
  • 0x5C: ADDR tied high (to VCC).

The sketch probes both and logs the detected address.

Measurement Mode

  • Continuous High‑Resolution is used. Typical first conversion ~120–180 ms.
  • readLightLevel() returns a float in lux, scaled by internal MTreg.

6) Payload Contract

The Flask route expects the following JSON keys. plant_id is an integer in this build.

{
  "sensor_type": "bh1750_lux",
  "reading": 123.45,     // float (lux)
  "plant_id": 1          // int
}

Server success responses are HTTP 200 or 201. Other codes are logged to Serial.

8) Runtime Behavior

State Flow

  1. Boot: Start Serial → init I²C → probe BH1750 (0x23 then 0x5C) → connect Wi‑Fi.
  2. Cycle (once per minute):
    1. Single high‑res lux read with readLightLevel().
    2. Clamp negative values to zero; treat NaN as invalid.
    3. POST JSON payload to Flask route.
  3. Connectivity: If Wi‑Fi drops, reconnect before next cycle.

Design Notes

  • No rolling average. One measurement per cycle reduces latency and server write volume.
  • Address probing supports both common module wirings (ADDR low or high).
  • 180 ms delay after begin ensures first valid conversion in continuous mode.

10) Full Sketch

/*
  ESP32 DevKit + BH1750 -> Flask JSON (every 60s)
  - Reads lux once per cycle using BH1750::readLightLevel()
  - Posts JSON: {"sensor_type": "...", "reading": <float>, "plant_id": <int>}
  - Pins: SDA=21, SCL=22 (per your hardware)
  - Library: "BH1750" by Christopher Laws (Library Manager)
*/

#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include <BH1750.h>

// ---------- USER CONFIG (edit these) ----------
const char* WIFI_SSID   = "4301b";   // e.g. "4301b"
const char* WIFI_PASS   = "scoutdog1";   // e.g. "scoutdog1"

const char* SERVER_URL  = "http://192.168.50.243:5001/add_data"; // your Flask route
const char* SENSOR_TYPE = "bh1750_lux";                           // keep as given
const int   PLANT_ID    = 1;                                      // adjust for plant

// Post once per minute
constexpr unsigned long POST_INTERVAL_MS = 60UL * 1000UL;

// I2C wiring for ESP32 DevKit (per your note)
constexpr uint8_t I2C_SDA = 21;
constexpr uint8_t I2C_SCL = 22;

// ---------- Globals ----------
BH1750 lightMeter;
unsigned long lastPostMs = 0;

// ---------- Forward decls ----------
void connectWiFi();
void ensureWiFiConnected();
bool initBH1750();
float readLuxOnce();
bool postReading(float lux);

// ---------- Setup ----------
void setup() {
  Serial.begin(115200);
  delay(150);
  Serial.println("\n[BOOT] ESP32 BH1750 -> Flask JSON (60s)");

  // Init I2C on the pins you specified
  Wire.begin(I2C_SDA, I2C_SCL);
  Serial.printf("[I2C] SDA=%u SCL=%u\n", I2C_SDA, I2C_SCL);

  // Bring up sensor. Try 0x23 then 0x5C to cover ADDR pin wiring.
  if (!initBH1750()) {
    Serial.println("[SENSOR] BH1750 init failed. Halting.");
    while (true) { delay(1000); }
  }

  // Wi-Fi
  connectWiFi();

  // Trigger immediate first POST
  lastPostMs = 0;
  Serial.println("[BOOT] Ready.");
}

// ---------- Loop ----------
void loop() {
  ensureWiFiConnected();

  unsigned long now = millis();
  if (lastPostMs == 0 || (now - lastPostMs) >= POST_INTERVAL_MS) {
    lastPostMs = now;

    // Single measurement per your requirement
    float lux = readLuxOnce();
    Serial.printf("[READ] lux=%.2f\n", lux);

    // Push to Flask route
    bool ok = postReading(lux);
    Serial.printf("[POST] %s\n", ok ? "OK" : "FAIL");
  }

  delay(50); // keep CPU cool, maintain responsiveness
}

// ---------- Wi-Fi helpers ----------
void connectWiFi() {
  Serial.printf("[WIFI] Connecting to SSID: %s\n", WIFI_SSID);
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  uint32_t start = millis();
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    if (millis() - start > 20000) { // 20s retry window
      Serial.println("\n[WIFI] Timeout. Re-attempting.");
      WiFi.disconnect(true);
      WiFi.begin(WIFI_SSID, WIFI_PASS);
      start = millis();
    }
  }
  Serial.printf("\n[WIFI] Connected. IP=%s\n", WiFi.localIP().toString().c_str());
}

void ensureWiFiConnected() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[WIFI] Disconnected. Reconnecting.");
    connectWiFi();
  }
}

// ---------- Sensor helpers ----------
bool initBH1750() {
  // Give sensor power-up time
  delay(150);

  // Try default 0x23 first
  if (lightMeter.begin(BH1750::CONTINUOUS_HIGH_RES_MODE, 0x23, &Wire)) {
    Serial.println("[SENSOR] BH1750 detected at 0x23");
    // Spec: first valid conversion ~120–180 ms in high-res
    delay(180);
    return true;
  }

  // Fallback to 0x5C if ADDR tied high
  if (lightMeter.begin(BH1750::CONTINUOUS_HIGH_RES_MODE, 0x5C, &Wire)) {
    Serial.println("[SENSOR] BH1750 detected at 0x5C");
    delay(180);
    return true;
  }

  Serial.println("[SENSOR] BH1750 not found on 0x23 or 0x5C. Check wiring/power.");
  return false;
}

float readLuxOnce() {
  // Simple single call per your request. No rolling average.
  float lux = lightMeter.readLightLevel(); // returns lux as float
  if (isnan(lux)) {
    Serial.println("[READ] NaN from sensor. Returning -1.");
    return -1.0f;
  }
  if (lux < 0.0f) lux = 0.0f; // clamp negatives
  return lux;
}

// ---------- HTTP POST ----------
bool postReading(float lux) {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[HTTP] No Wi-Fi. Abort.");
    return false;
  }

  WiFiClient client;  // plain HTTP
  HTTPClient http;
  http.setTimeout(5000); // 5s

  if (!http.begin(client, SERVER_URL)) {
    Serial.println("[HTTP] http.begin() failed");
    return false;
  }

  http.addHeader("Content-Type", "application/json");

  // Flask expects: sensor_type (string), reading (float), plant_id (int/string)
  String payload = String("{\"sensor_type\":\"") + SENSOR_TYPE +
                   "\",\"reading\":" + String(lux, 2) +
                   ",\"plant_id\":" + String(PLANT_ID) + "}";

  Serial.print("[HTTP] POST payload: ");
  Serial.println(payload);

  int code = http.POST(payload);
  Serial.printf("[HTTP] Status: %d\n", code);

  String resp = http.getString();
  Serial.print("[HTTP] Response: ");
  Serial.println(resp);

  http.end();
  return (code == 201 || code == 200);
}

12) Troubleshooting

No Sensor Detected

  • Run an I²C scanner to confirm address (0x23 or 0x5C).
  • Check SDA/SCL orientation; swap if needed.
  • Verify ADDR pin state matches expected address.

Unstable Lux Values

  • Shield sensor from direct LED flicker; try diffusing cover.
  • Use continuous high‑res mode; avoid rapid re‑init.
  • Optionally median‑filter on server side.

Wi‑Fi / HTTP Failures

  • Check credentials and 2.4 GHz availability.
  • Ping the Flask host from another device.
  • Inspect server logs for request errors.

Units & Scaling

  • BH1750 outputs lux directly; no extra scaling needed with standard MTreg.
  • Clamp negative or NaN readings as the sketch does.

13) Validation Checklist

  • I²C scanner finds BH1750 at 0x23 or 0x5C.
  • Serial shows successful Wi‑Fi connection.
  • One HTTP 200/201 per minute.
  • Database row contains sensor_type="bh1750_lux" and plausible lux value.

14) Security & Hardening

  • Using plain HTTP on LAN. For WAN, put Flask behind TLS.
  • Add a shared token header (e.g., X-API-Key) and verify server‑side.
  • Rate‑limit or authenticate if deploying multiple nodes.

15) Extensions & Variations

  • Adjust sensitivity via MTreg if your library exposes it (e.g., darker environments).
  • Batch readings or include device metadata in payload if schema evolves.
  • Sleep ESP32 between cycles to lower power draw.
  • Combine with DHTxx or other I²C sensors on the same bus.
